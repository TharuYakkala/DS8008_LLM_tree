def predict_heart_disease(
    age, sex, chest_pain_type, resting_blood_pressure, serum_cholesterol,
    fasting_blood_sugar, resting_ecg_results, maximum_heart_rate_achieved,
    exercise_induced_angina, st_depression_exercise,
    slope_peak_exercise_st, num_major_vessels, thalassemia
):
    node_truth_values = []

    # Node 1: Root Node - Chest Pain Type (Primary symptom indicator)
    # Condition: Is the chest pain typical or atypical angina?
    if chest_pain_type in ['Typical angina', 'Atypical angina']:
        node_truth_values.append(1) # Condition of Node 1 is True
        # Branch for more severe chest pain types
        # Node 2 (child of Node 1, True branch): Exercise Induced Angina (confirmatory symptom)
        # Condition: Is angina induced by exercise?
        if exercise_induced_angina == 'yes':
            node_truth_values.append(1) # Condition of Node 2 is True
            return 1, node_truth_values # Prediction: Heart Disease (strong symptomatic evidence)
        else:
            node_truth_values.append(0) # Condition of Node 2 is False
            return 0, node_truth_values # Prediction: No Heart Disease (symptoms present, but no exercise-induced angina, suggesting lower cardiac likelihood or other cause)
    else:
        node_truth_values.append(0) # Condition of Node 1 is False
        # Branch for less severe chest pain types (Non-anginal or Asymptomatic)
        # Node 3 (child of Node 1, False branch): Thalassemia (structural defect indicator)
        # Condition: Are there thalassemia defects (fixed or reversible)?
        if thalassemia in ['Fixed defect', 'Reversible defect']:
            node_truth_values.append(1) # Condition of Node 3 is True
            return 1, node_truth_values # Prediction: Heart Disease (even if symptoms are mild, structural defect is a strong indicator)
        else:
            node_truth_values.append(0) # Condition of Node 3 is False
            return 0, node_truth_values # Prediction: No Heart Disease (no severe pain, no underlying structural defect in thalassemia)
