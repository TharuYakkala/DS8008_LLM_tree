def classify_party(features):
    """
    Classifies party affiliation (Democrat or Republican) based on voting records.

    This decision tree is induced based on general political knowledge of the 1980s US Congress,
    prioritizing features that historically showed a strong partisan divide.
    It has a maximum depth of 2.

    Args:
        features (dict): A dictionary where keys are feature names (vote topics)
                         and values are 'yes', 'no', or None (if missing).

    Returns:
        tuple: A tuple containing:
               - str: The predicted party affiliation ('Democrat' or 'Republican').
               - list: A list representing the truth values of the inner nodes encountered
                       along the decision path. 1 if the condition is satisfied, 0 otherwise.

    inner_nodes_truth_values = []
    prediction = None

    # Node 1 (Root): Vote on aid to Nicaraguan contras
    # Condition: Is the vote 'yes'? (Voting 'yes' on Contra aid was a strong Republican stance)
    # This is often the most polarizing foreign policy issue of the era.
    is_contras_yes = (features.get('Vote on aid to Nicaraguan contras') == 'yes')
    inner_nodes_truth_values.append(1 if is_contras_yes else 0)

    if is_contras_yes:
        # Path for 'yes' on Contra aid: strongly leans Republican.
        # To refine, use a domestic spending issue.

        # Node 2 (Left child of Node 1): Vote on education spending
        # Condition: Is the vote 'no'? (Republicans generally favored less federal education spending)
        is_education_no = (features.get('Vote on education spending') == 'no')
        inner_nodes_truth_values.append(1 if is_education_no else 0)

        if is_education_no:
            # 'yes' on Contras AND 'no' on education spending: Very strong Republican signal.
            prediction = "Republican"
        else:
            # 'yes' on Contras AND 'yes' (or missing) on education spending:
            # Still highly likely Republican due to the strong Contra vote,
            # but might indicate a more moderate or fiscally less conservative Republican.
            prediction = "Republican"
    else:
        # Path for 'no' (or missing) on Contra aid: strongly leans Democrat.
        # To refine, use a social/cultural issue.

        # Node 3 (Right child of Node 1): Vote on allowing religious groups in schools
        # Condition: Is the vote 'no'? (Democrats generally opposed allowing religious groups in schools,
        # often citing church-state separation)
        is_religious_no = (features.get('Vote on allowing religious groups in schools') == 'no')
        inner_nodes_truth_values.append(1 if is_religious_no else 0)

        if is_religious_no:
            # 'no' on Contras AND 'no' on religious groups in schools: Very strong Democrat signal.
            prediction = "Democrat"
        else:
            # 'no' on Contras AND 'yes' (or missing) on religious groups in schools:
            # Still highly likely Democrat due to the strong anti-Contra vote,
            # but might indicate a more conservative Democrat or one who supports local religious expression.
            prediction = "Democrat"

    return prediction, inner_nodes_truth_values
