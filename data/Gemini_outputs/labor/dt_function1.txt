def classify_labor_negotiation(
    duration_of_agreement: int,
    wage_increase_in_first_year_of_contract: int,
    wage_increase_in_second_year_of_contract: int,
    wage_increase_in_third_year_of_contract: int,
    cost_of_living_adjustment: str,
    number_of_working_hours_during_week: int,
    employer_contributions_to_pension_plan: str,
    standby_pay: int,
    shift_differencial_supplement_for_work_on_2_and_3_shift: int,
    education_allowance: str,
    number_of_statutory_holidays: int,
    number_of_paid_vacation_days: str,
    employer_s_help_during_employee_longterm_disability: str,
    employers_contribution_towards_the_dental_plan: str,
    employer_s_financial_contribution_towards_the_covering_the_costs_of_bereavement: str,
    employer_s_contribution_towards_the_health_plan: str
) -> (str, list[int]):
    """
    Classifies the outcome of labor negotiations as 'good' or 'bad' based on key features.

    Args:
        duration_of_agreement (int): Duration of the agreement (1 to 7).
        wage_increase_in_first_year_of_contract (int): Wage increase in the first year (2 to 7).
        wage_increase_in_second_year_of_contract (int): Wage increase in the second year (2 to 7).
        wage_increase_in_third_year_of_contract (int): Wage increase in the third year (2 to 7).
        cost_of_living_adjustment (str): Type of cost of living adjustment (none / temporary cost of living adjustment formula / total compensation adjustment).
        number_of_working_hours_during_week (int): Number of working hours per week (35 to 40).
        employer_contributions_to_pension_plan (str): Employer's pension plan contributions (none / retirement allowance included / employee contribution).
        standby_pay (int): Standby pay (2 to 25).
        shift_differencial_supplement_for_work_on_2_and_3_shift (int): Shift differential supplement (1 to 25).
        education_allowance (str): Education allowance (yes / no).
        number_of_statutory_holidays (int): Number of statutory holidays (9 to 15).
        number_of_paid_vacation_days (str): Number of paid vacation days (below average / average / generous).
        employer_s_help_during_employee_longterm_disability (str): Employer's help during long-term disability (yes / no).
        employers_contribution_towards_the_dental_plan (str): Employer's dental plan contribution (none / half / full).
        employer_s_financial_contribution_towards_the_covering_the_costs_of_bereavement (str): Employer's bereavement financial contribution (yes / no).
        employer_s_contribution_towards_the_health_plan (str): Employer's health plan contribution (none / half / full).

    Returns:
        (str, list[int]): A tuple containing the prediction ('good' or 'bad')
                         and a list of truth values for the conditions of the inner nodes traversed.
                         1 if the condition is satisfied, 0 otherwise.
    """
    truth_values = []

    # Node 1 (Root Split): High immediate wage increase is a primary indicator of a good outcome.
    condition_node1 = (wage_increase_in_first_year_of_contract > 4)
    truth_values.append(1 if condition_node1 else 0)

    if condition_node1:
        # Path: Wage increase in first year is high.
        # Node 2.1 (Child Split): Evaluate health benefits as a secondary factor for a good outcome.
        condition_node2_1 = (employer_s_contribution_towards_the_health_plan == 'full')
        truth_values.append(1 if condition_node2_1 else 0)

        # With a high initial wage increase, the outcome is generally considered 'good'
        # within the depth-2 constraint, regardless of the health plan being fully covered or not.
        prediction = 'good'
    else:
        # Path: Wage increase in first year is low or medium.
        # Node 2.2 (Child Split): A strong Cost of Living Adjustment can compensate for lower immediate wage gains.
        condition_node2_2 = (cost_of_living_adjustment == 'total compensation adjustment')
        truth_values.append(1 if condition_node2_2 else 0)

        if condition_node2_2:
            prediction = 'good'  # Total compensation adjustment makes it good despite lower initial wage.
        else:
            # If wage increase is not high and there's no strong COLA, the outcome is generally 'bad'.
            prediction = 'bad'

    return prediction, truth_values
