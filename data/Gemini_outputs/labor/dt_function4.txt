def classify_labor_negotiations(
    duration_of_agreement,
    wage_increase_in_first_year_of_contract,
    wage_increase_in_second_year_of_contract,
    wage_increase_in_third_year_of_contract,
    cost_of_living_adjustment,
    number_of_working_hours_during_week,
    employer_contributions_to_pension_plan,
    standby_pay,
    shift_differencial_supplement_for_work_on_2_and_3_shift,
    education_allowance,
    number_of_statutory_holidays,
    number_of_paid_vacation_days,
    employer_s_help_during_employee_longterm_disability,
    employers_contribution_towards_the_dental_plan,
    employer_s_financial_contribution_towards_the_covering_the_costs_of_bereavement,
    employer_s_contribution_towards_the_health_plan
):
    inner_node_truth_values = []
    prediction = None

    # Node 0 (Root): Is the first-year wage increase substantial?
    # A wage increase greater than 4 (meaning 5, 6, or 7) is considered a strong positive sign.
    if wage_increase_in_first_year_of_contract > 4:
        inner_node_truth_values.append(1)  # Condition for Node 0 is satisfied
        prediction = "good"
    else:
        inner_node_truth_values.append(0)  # Condition for Node 0 is not satisfied
        # Node 1: If the wage increase is not substantial, is the health plan contribution excellent?
        # A full employer's contribution to the health plan is a major benefit that can still indicate a good outcome.
        if employer_s_contribution_towards_the_health_plan == "full":
            inner_node_truth_values.append(1)  # Condition for Node 1 is satisfied
            prediction = "good"
        else:
            inner_node_truth_values.append(0)  # Condition for Node 1 is not satisfied
            # If both wage increase is not substantial and health plan is not fully covered, the outcome is likely bad.
            prediction = "bad"

    return prediction, inner_node_truth_values
