def classify_leaving_certificate(sex, drumcondra_score, educational_level, father_prestige_score, school_type):
    """
    Classifies whether a school leaving certificate is taken based on provided features.
    This decision tree has a maximum depth of 2.

    Args:
        sex (str): 'male' or 'female'.
        drumcondra_score (int): Drumcondra Verbal Reasoning Test Score.
        educational_level (int): Educational level attained (0 to 10).
        father_prestige_score (int): Prestige score for father's occupation (0 to 28).
        school_type (str): Type of school ('secondary', 'vocational', 'primary terminal leaver').

    Returns:
        tuple: A tuple containing:
            - str: The predicted class ('Yes' or 'No' for taking the leaving certificate).
            - list: A list of integers (0 or 1) representing the truth values of conditions
                    at each inner node traversed. 1 if the condition is satisfied, 0 otherwise.
    """
    truth_values = []

    # Root Node: Primary determinant for educational pathway is the type of school.
    # Condition 1: Is the student designated as a 'primary terminal leaver'?
    # This path strongly indicates not pursuing a leaving certificate.
    if school_type == 'primary terminal leaver':
        truth_values.append(1) # Condition 1: school_type == 'primary terminal leaver' is True
        return "No", truth_values
    else:
        truth_values.append(0) # Condition 1: school_type == 'primary terminal leaver' is False

        # Node 2: If not a primary terminal leaver, differentiate between 'secondary' and 'vocational' paths.
        # Condition 2: Is the student in a 'secondary' school?
        if school_type == 'secondary':
            truth_values.append(1) # Condition 2: school_type == 'secondary' is True
            # Node 3 (Depth 2 split for secondary students): Verbal reasoning score is key for academic success.
            # Condition 3: Is the Drumcondra Verbal Reasoning Test Score high? (e.g., > 60)
            if drumcondra_score > 60:
                truth_values.append(1) # Condition 3: drumcondra_score > 60 is True
                return "Yes", truth_values # High academic potential in secondary school -> likely takes certificate
            else:
                truth_values.append(0) # Condition 3: drumcondra_score > 60 is False
                return "No", truth_values # Lower academic potential in secondary school -> might drop out or not take certificate
        # Node 4 (Depth 1 split: Implicitly, if not primary terminal and not secondary, then vocational)
        # Condition 4 (Implicit): school_type == 'vocational' (represented by condition 2 being false for 'secondary')
        else: # school_type must be 'vocational'
            truth_values.append(0) # Condition 2: school_type == 'secondary' is False
            # Node 5 (Depth 2 split for vocational students): Father's occupation prestige can indicate family support/aspirations.
            # Condition 5: Is the father's occupation prestige score high? (e.g., > 15)
            if father_prestige_score > 15:
                truth_values.append(1) # Condition 4: father_prestige_score > 15 is True
                return "Yes", truth_values # Vocational students with higher socioeconomic background -> might pursue academic cert
            else:
                truth_values.append(0) # Condition 4: father_prestige_score > 15 is False
                return "No", truth_values # Vocational students with lower socioeconomic background -> less likely to pursue academic cert
