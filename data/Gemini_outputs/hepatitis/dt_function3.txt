def classify_survival(age, sex, steroids, antivirals, fatigue, malaise, anorexia, liver_big, liver_firm, spleen_palpable, spider_angiomata, ascites, varices, bilirubin, alkaline_phosphatase, serum_glutamic_oxaloacetic_transaminase_level, albumin_level, prothrombin_time, histologic_data):
    """
    Classifies whether a patient survives or dies based on key clinical and lab features.
    
    The decision tree has a maximum depth of 2, using prothrombin time and albumin level
    as the most critical prognostic indicators.

    Args:
        age (int): Patient's age in years.
        sex (str): Patient's sex ('male' or 'female').
        steroids (str): Whether steroids were administered ('yes' or 'no').
        antivirals (str): Whether antivirals were administered ('yes' or 'no').
        fatigue (str): Presence of fatigue ('yes' or 'no').
        malaise (str): Presence of malaise ('yes' or 'no').
        anorexia (str): Presence of anorexia ('yes' or 'no').
        liver_big (str): Whether liver is enlarged ('yes' or 'no').
        liver_firm (str): Whether liver is firm ('yes' or 'no').
        spleen_palpable (str): Whether spleen is palpable ('yes' or 'no').
        spider_angiomata (str): Presence of spider angiomata ('yes' or 'no').
        ascites (str): Presence of ascites ('yes' or 'no').
        varices (str): Presence of varices ('yes' or 'no').
        bilirubin (float): Bilirubin level in mg/dL.
        alkaline_phosphatase (float): Alkaline phosphatase level in units/liter.
        serum_glutamic_oxaloacetic_transaminase_level (float): SGOT level in units/liter.
        albumin_level (float): Albumin level in g/dL.
        prothrombin_time (float): Prothrombin time in seconds.
        histologic_data (str): Whether histologic data is available ('yes' or 'no').

    Returns:
        tuple: A tuple containing:
            - str: The predicted outcome ('survives' or 'dies').
            - list: A list of integers (0 or 1) representing the truth values of the inner nodes.
                    1 if the condition of the corresponding inner node is satisfied, 0 otherwise.
    """
    node_truth_values = []

    # Node 1 (Root): Split on Prothrombin time (seconds)
    # Prothrombin time is a critical indicator of the liver's synthetic function and clotting ability.
    # Significantly prolonged prothrombin time (e.g., >16 seconds) indicates severe liver dysfunction
    # and is strongly associated with poor prognosis.
    if prothrombin_time > 16.0:
        node_truth_values.append(1)  # Condition of the first inner node is satisfied
        # High prothrombin time often leads to a "dies" prediction.
        prediction = "dies"
    else:
        node_truth_values.append(0)  # Condition of the first inner node is NOT satisfied
        # Node 2: Split on Albumin level (g/dL)
        # Albumin is also synthesized by the liver, and low levels indicate impaired liver function.
        # An albumin level <= 3.0 g/dL is considered clinically low and suggests chronic or severe liver disease.
        if albumin_level <= 3.0:
            node_truth_values.append(1)  # Condition of the second inner node is satisfied
            prediction = "dies"
        else:
            node_truth_values.append(0)  # Condition of the second inner node is NOT satisfied
            # If both prothrombin time and albumin levels are relatively within acceptable ranges,
            # the patient is predicted to survive based on these primary indicators.
            prediction = "survives"

    return prediction, node_truth_values
