def classify_patient_survival(
    age_years: int,
    sex: str, # 'male' / 'female'
    steroids: str, # 'yes' / 'no'
    antivirals: str, # 'yes' / 'no'
    fatigue: str, # 'yes' / 'no'
    malaise: str, # 'yes' / 'no'
    anorexia: str, # 'yes' / 'no'
    liver_big: str, # 'yes' / 'no'
    liver_firm: str, # 'yes' / 'no'
    spleen_palpable: str, # 'yes' / 'no'
    spider_angiomata: str, # 'yes' / 'no'
    ascites: str, # 'yes' / 'no'
    varices: str, # 'yes' / 'no'
    bilirubin_mg_dL: float,
    alkaline_phosphatase_units_liter: float,
    serum_glutamic_oxaloacetic_transaminase_level_units_liter: float,
    albumin_level_g_dL: float,
    prothrombin_time_seconds: float,
    histologic_data: str # 'yes' / 'no'
) -> tuple[str, list[int]]:
    """
    Classifies patient survival (survives/dies) based on medical features using a decision tree.

    The decision tree has a maximum depth of 2 and uses the most prognostically significant features
    for general medical conditions, particularly those affecting the liver.

    Args:
        age_years (int): Patient's age in years.
        sex (str): Patient's sex ('male' or 'female').
        steroids (str): Whether steroids were administered ('yes' or 'no').
        antivirals (str): Whether antivirals were administered ('yes' or 'no').
        fatigue (str): Presence of fatigue ('yes' or 'no').
        malaise (str): Presence of malaise ('yes' or 'no').
        anorexia (str): Presence of anorexia ('yes' or 'no').
        liver_big (str): Whether the liver is enlarged ('yes' or 'no').
        liver_firm (str): Whether the liver is firm ('yes' or 'no').
        spleen_palpable (str): Whether the spleen is palpable ('yes' or 'no').
        spider_angiomata (str): Presence of spider angiomata ('yes' or 'no').
        ascites (str): Presence of ascites ('yes' or 'no').
        varices (str): Presence of varices ('yes' or 'no').
        bilirubin_mg_dL (float): Bilirubin level in mg/dL.
        alkaline_phosphatase_units_liter (float): Alkaline phosphatase level in units/liter.
        serum_glutamic_oxaloacetic_transaminase_level_units_liter (float): SGOT level in units/liter.
        albumin_level_g_dL (float): Albumin level in g/dL.
        prothrombin_time_seconds (float): Prothrombin time in seconds.
        histologic_data (str): Whether histologic data is available ('yes' or 'no').

    Returns:
        tuple[str, list[int]]: A tuple containing the predicted outcome ('survives' or 'dies')
                               and a list of truth values for the inner nodes encountered.
                               Each truth value is 1 if the condition is met, 0 otherwise.
    """
    prediction = ""
    inner_node_truth_values = []

    # Node 1: Check Prothrombin time (a key indicator of liver synthetic function and coagulation)
    # A significantly prolonged prothrombin time (e.g., > 16.0 seconds) indicates severe liver dysfunction.
    if prothrombin_time_seconds > 16.0:
        inner_node_truth_values.append(1) # Condition met
        prediction = "dies"
    else:
        inner_node_truth_values.append(0) # Condition not met
        # Node 2: If prothrombin time is not severely prolonged, check Albumin level
        # Low albumin (e.g., <= 3.0 g/dL) suggests poor liver synthetic function or malnutrition,
        # which can also indicate a poor prognosis.
        if albumin_level_g_dL <= 3.0:
            inner_node_truth_values.append(1) # Condition met
            prediction = "dies"
        else:
            inner_node_truth_values.append(0) # Condition not met
            prediction = "survives"

    return prediction, inner_node_truth_values
