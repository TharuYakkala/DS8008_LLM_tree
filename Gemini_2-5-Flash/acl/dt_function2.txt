def predict_cartilage_change(
    group,
    sex,
    height_cm,
    body_mass_kg,
    body_mass_index_kg_m2,
    age_years,
    Lysholm_score,
    IKDC_score,
    Tegner_score,
    baseline_mean_cartilage_thickness_in_the_central_medial_femur_mm,
    baseline_mean_T2_values_for_the_deep_cartilage_layers_in_the_central_medial_femur_ms,
    baseline_mean_T2_values_for_the_superficial_cartilage_layers_in_the_central_medial_femur_ms,
    external_peak_knee_moment_around_x_axis_Nm,
    external_peak_knee_moment_around_y_axis_Nm,
    external_peak_knee_moment_around_z_axis_Nm,
    dominant_leg,
    days_from_injury_to_baseline_MRI
):
    """
    Predicts whether cartilage thickness change in the central medial femur over 12 months
    following ACL injury is normal or abnormal.

    Parameters based on provided features:
    - group (str): Patient group ('coper', 'noncoper', 'surgical reconstruction')
    - sex (str): 'female' or 'male'
    - height_cm (float): Height in cm
    - body_mass_kg (float): Body mass in kg
    - body_mass_index_kg_m2 (float): Body mass index in kg/m^2
    - age_years (int): Age in years
    - Lysholm_score (int): Lysholm score
    - IKDC_score (int): IKDC score
    - Tegner_score (int): Tegner score
    - baseline_mean_cartilage_thickness_in_the_central_medial_femur_mm (float): Baseline thickness in mm
    - baseline_mean_T2_values_for_the_deep_cartilage_layers_in_the_central_medial_femur_ms (float): Baseline T2 in ms for deep layers
    - baseline_mean_T2_values_for_the_superficial_cartilage_layers_in_the_central_medial_femur_ms (float): Baseline T2 in ms for superficial layers
    - external_peak_knee_moment_around_x_axis_Nm (float): Peak knee moment around x-axis in Nm
    - external_peak_knee_moment_around_y_axis_Nm (float): Peak knee moment around y-axis in Nm
    - external_peak_knee_moment_around_z_axis_Nm (float): Peak knee moment around z-axis in Nm
    - dominant_leg (str): Dominant leg
    - days_from_injury_to_baseline_MRI (int): Days from injury to baseline MRI

    Returns:
    - prediction (str): 'normal' or 'abnormal'
    - inner_nodes_truth (list): A list of 0s and 1s representing the truth values
                                 of the inner nodes in breadth-first order.
                                 [Root Node (T2 split), Left Child (BMI split), Right Child (Group split)]
    """
    inner_nodes_truth = [0] * 3  # Represents conditions for 3 potential splits (Root, Left Child, Right Child)

    # Alias for chosen features for brevity
    baseline_T2_deep = baseline_mean_T2_values_for_the_deep_cartilage_layers_in_the_central_medial_femur_ms
    bmi = body_mass_index_kg_m2

    # Node 0 (Root Node): Split based on baseline deep cartilage T2 values
    # Higher T2 values often indicate existing cartilage matrix degradation.
    if baseline_T2_deep > 42.0:  # Hypothetical threshold for elevated T2
        inner_nodes_truth[0] = 1  # Root condition: baseline_T2_deep > 42.0 is True
        # Path for elevated baseline T2 values (already compromised cartilage)
        
        # Node 2 (Right Child of Root): Split based on surgical reconstruction
        # Surgical reconstruction might mitigate further degradation even if T2 is high.
        if group == 'surgical reconstruction':
            inner_nodes_truth[2] = 1  # Right child condition: group == 'surgical reconstruction' is True
            prediction = 'normal'  # Stabilized joint, potentially normal change
        else:
            inner_nodes_truth[2] = 0  # Right child condition: group == 'surgical reconstruction' is False
            prediction = 'abnormal' # Existing damage + no reconstruction -> likely abnormal change
    else:
        inner_nodes_truth[0] = 0  # Root condition: baseline_T2_deep > 42.0 is False
        # Path for relatively healthy baseline T2 values
        
        # Node 1 (Left Child of Root): Split based on Body Mass Index
        # High BMI contributes to increased mechanical loading on the joint.
        if bmi > 27.0:  # Hypothetical threshold for overweight/mildly obese
            inner_nodes_truth[1] = 1  # Left child condition: body_mass_index_kg_m2 > 27.0 is True
            prediction = 'abnormal'  # Healthy T2, but high BMI increases risk of abnormal change
        else:
            inner_nodes_truth[1] = 0  # Left child condition: body_mass_index_kg_m2 > 27.0 is False
            prediction = 'normal'  # Healthy T2, healthy BMI -> likely normal change

    return prediction, inner_nodes_truth
