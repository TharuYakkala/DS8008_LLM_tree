def classify_negotiation_outcome(duration_of_agreement, wage_increase_in_first_year_of_contract, wage_increase_in_second_year_of_contract, wage_increase_in_third_year_of_contract, cost_of_living_adjustment, number_of_working_hours_during_week, employer_contributions_to_pension_plan, standby_pay, shift_differencial_supplement_for_work_on_2_and_3_shift, education_allowance, number_of_statutory_holidays, number_of_paid_vacation_days, employer_s_help_during_employee_longterm_disability, employers_contribution_towards_the_dental_plan, employer_s_financial_contribution_towards_the_covering_the_costs_of_bereavement, employer_s_contribution_towards_the_health_plan):
    """
    Classifies the outcome of labor negotiations as 'good' or 'bad' based on key features.
    
    Args:
        duration_of_agreement (int): Duration of the agreement (1 to 7).
        wage_increase_in_first_year_of_contract (int): Wage increase in the first year (2 to 7).
        wage_increase_in_second_year_of_contract (int): Wage increase in the second year (2 to 7).
        wage_increase_in_third_year_of_contract (int): Wage increase in the third year (2 to 7).
        cost_of_living_adjustment (str): Type of cost of living adjustment ('none', 'temporary cost of living adjustment formula', 'total compensation adjustment').
        number_of_working_hours_during_week (int): Number of working hours (35 to 40).
        employer_contributions_to_pension_plan (str): Pension plan contribution ('none', 'retirement allowance included', 'employee contribution').
        standby_pay (int): Standby pay (2 to 25).
        shift_differencial_supplement_for_work_on_2_and_3_shift (int): Shift differential (1 to 25).
        education_allowance (str): Education allowance ('yes', 'no').
        number_of_statutory_holidays (int): Number of statutory holidays (9 to 15).
        number_of_paid_vacation_days (str): Number of paid vacation days ('below average', 'average', 'generous').
        employer_s_help_during_employee_longterm_disability (str): Help during long-term disability ('yes', 'no').
        employers_contribution_towards_the_dental_plan (str): Dental plan contribution ('none', 'half', 'full').
        employer_s_financial_contribution_towards_the_covering_the_costs_of_bereavement (str): Bereavement costs contribution ('yes', 'no').
        employer_s_contribution_towards_the_health_plan (str): Health plan contribution ('none', 'half', 'full').

    Returns:
        tuple: A tuple containing the classification ('good' or 'bad') and a list of truth values 
               for the inner nodes traversed (1 if condition satisfied, 0 otherwise).
    """
    truth_values = []

    # Root Node Split: Is the employer's contribution towards the health plan 'full'?
    # Health plan contribution is a critical indicator of employee benefits.
    if employer_s_contribution_towards_the_health_plan == 'full':
        truth_values.append(1)  # Condition 'employer_s_contribution_towards_the_health_plan == "full"' is True
        # Node 1.1: If full health plan, is the first-year wage increase at least moderate (>=4)?
        # A full health plan is very beneficial, but a very low wage increase can still make the outcome less good.
        if wage_increase_in_first_year_of_contract >= 4:
            truth_values.append(1)  # Condition 'wage_increase_in_first_year_of_contract >= 4' is True
            return 'good', truth_values
        else:
            truth_values.append(0)  # Condition 'wage_increase_in_first_year_of_contract >= 4' is False
            return 'bad', truth_values # Full health, but very low wage increase (2 or 3) makes it a bad outcome.
    else:  # employer_s_contribution_towards_the_health_plan is 'none' or 'half'
        truth_values.append(0)  # Condition 'employer_s_contribution_towards_the_health_plan == "full"' is False
        # Node 1.2: If health plan isn't full, does a substantial first-year wage increase (>=5) compensate?
        # A high wage increase is needed to offset the lack of full health benefits.
        if wage_increase_in_first_year_of_contract >= 5:
            truth_values.append(1)  # Condition 'wage_increase_in_first_year_of_contract >= 5' is True
            return 'good', truth_values
        else:
            truth_values.append(0)  # Condition 'wage_increase_in_first_year_of_contract >= 5' is False
            return 'bad', truth_values # Neither full health nor a substantial wage increase leads to a bad outcome.
