def predict_party_affiliation(
    handicapped_infants_vote: str,
    water_project_vote: str,
    budget_resolution_vote: str,
    physician_fee_freeze_vote: str,
    el_salvador_aid_vote: str,
    religious_groups_in_schools_vote: str,
    anti_satellite_test_ban_vote: str,
    nicaraguan_contras_aid_vote: str,
    mx_missile_vote: str,
    immigration_vote: str,
    synfuels_cutback_vote: str,
    education_spending_vote: str,
    superfund_right_to_sue_vote: str,
    crime_vote: str,
    duty_free_exports_vote: str,
    export_admin_south_africa_vote: str
) -> (str, list[int]):
    """
    Classifies party affiliation (Democrat or Republican) based on voting records.

    The decision tree is constructed based on a maximum depth of 2,
    prioritizing features historically most indicative of party lines
    in the context of the provided issues (typical of 1980s US politics).

    Parameters (all are 'yes' or 'no' strings):
    - handicapped_infants_vote
    - water_project_vote
    - budget_resolution_vote
    - physician_fee_freeze_vote
    - el_salvador_aid_vote
    - religious_groups_in_schools_vote
    - anti_satellite_test_ban_vote
    - nicaraguan_contras_aid_vote
    - mx_missile_vote
    - immigration_vote
    - synfuels_cutback_vote
    - education_spending_vote
    - superfund_right_to_sue_vote
    - crime_vote
    - duty_free_exports_vote
    - export_admin_south_africa_vote

    Returns:
    - prediction (str): "Democrat" or "Republican"
    - inner_node_truth_values (list[int]): A list of 0s and 1s representing
      whether the condition of each inner node is satisfied.
      The order corresponds to:
      1. Root node: `nicaraguan_contras_aid_vote == "yes"`
      2. Left child node (if root condition is true): `mx_missile_vote == "yes"`
      3. Right child node (if root condition is false): `export_admin_south_africa_vote == "yes"`
    """

    inner_node_truth_values = []
    prediction = None

    # Node 1 (Root): Vote on aid to Nicaraguan contras
    # This was a highly polarizing foreign policy issue during the 1980s,
    # with Republicans largely supporting the Contras and Democrats largely opposing.
    condition_node1 = (nicaraguan_contras_aid_vote == "yes")
    inner_node_truth_values.append(1 if condition_node1 else 0)

    # Node 2 (Left Child): Vote on MX missile (for the "yes" branch of Node 1)
    # The MX missile program was a key part of Reagan's military buildup,
    # generally supported by Republicans and opposed by Democrats.
    condition_node2 = (mx_missile_vote == "yes")
    inner_node_truth_values.append(1 if condition_node2 else 0)

    # Node 3 (Right Child): Vote on Export Administration Act - South Africa (for the "no" branch of Node 1)
    # Sanctions against apartheid South Africa were strongly supported by Democrats,
    # while Republicans were more divided, with many opposing sanctions.
    condition_node3 = (export_admin_south_africa_vote == "yes")
    inner_node_truth_values.append(1 if condition_node3 else 0)

    # Decision Logic (max_depth=2)
    if condition_node1:  # nicaraguan_contras_aid_vote == "yes" (Strong Republican indicator)
        if condition_node2:  # mx_missile_vote == "yes" (Further reinforces Republican stance)
            prediction = "Republican"
        else:  # mx_missile_vote == "no" (Still leans Republican, as Contras vote is a strong predictor)
            prediction = "Republican"
    else:  # nicaraguan_contras_aid_vote == "no" (Strong Democrat indicator)
        if condition_node3:  # export_admin_south_africa_vote == "yes" (Further reinforces Democrat stance)
            prediction = "Democrat"
        else:  # export_admin_south_africa_vote == "no" (Still leans Democrat, as Contras vote is a strong predictor)
            prediction = "Democrat"

    return prediction, inner_node_truth_values
