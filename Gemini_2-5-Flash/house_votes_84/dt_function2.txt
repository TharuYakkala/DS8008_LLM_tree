def classify_party(features):
    """
    Classifies party affiliation (Democrat or Republican) based on voting records.
    The decision tree has a maximum depth of 2.

    Args:
        features (dict): A dictionary where keys are feature names (str) and
                         values are 'yes' or 'no' (str) representing a vote.

    Returns:
        tuple: A tuple containing:
            - str: The predicted party affiliation ('Democrat' or 'Republican').
            - list: A list of 0s and 1s representing the truth values of the
                    inner nodes. The nodes are ordered as follows:
                    [Root Node, Left Child of Root, Right Child of Root].
                    1 if the condition is satisfied, 0 otherwise.
    """
    # Initialize truth values for up to 3 inner nodes in a depth-2 tree:
    # [Node 0 (Root), Node 1 (Left child of Root), Node 2 (Right child of Root)]
    inner_node_truth_values = [0, 0, 0]

    # Node 0: Vote on aid to Nicaraguan contras (Root Node)
    # This is historically one of the most polarizing votes, strongly differentiating parties.
    if features['Vote on aid to Nicaraguan contras'] == 'yes':
        inner_node_truth_values[0] = 1  # Root node condition satisfied
        # If 'yes' on Contras, this is a strong Republican leaning.
        # Node 1: Vote on physician fee freeze (Left child of Root)
        # This vote also had significant party-line divisions.
        if features['Vote on physician fee freeze'] == 'yes':
            inner_node_truth_values[1] = 1  # Left child condition satisfied
            prediction = 'Republican'
        else:  # 'no' on physician fee freeze
            inner_node_truth_values[1] = 0  # Left child condition not satisfied
            # Even if 'no' on fee freeze (a somewhat Democratic stance),
            # the 'yes' on Contras is a dominant Republican indicator for this era.
            prediction = 'Republican'
    else:  # 'no' on aid to Nicaraguan contras
        inner_node_truth_values[0] = 0  # Root node condition not satisfied
        # If 'no' on Contras, this is a strong Democrat leaning.
        # Node 2: Vote on Export Administration Act - South Africa (Right child of Root)
        # Sanctions against apartheid South Africa were generally supported by Democrats.
        if features['Vote on Export Administration Act - South Africa'] == 'yes':
            inner_node_truth_values[2] = 1  # Right child condition satisfied
            prediction = 'Democrat'
        else:  # 'no' on Export Administration Act - South Africa
            inner_node_truth_values[2] = 0  # Right child condition not satisfied
            # If 'no' on Contras (Democrat leaning) AND 'no' on South Africa sanctions (Republican leaning, opposing sanctions),
            # the 'no' on SA sanctions is a strong Republican deviation for this branch,
            # indicating a more conservative stance despite the Contras vote.
            prediction = 'Republican'

    return prediction, inner_node_truth_values
