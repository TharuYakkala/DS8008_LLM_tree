def classify_party(features):
    """
    Classifies party affiliation (Democrat or Republican) based on voting records.

    Parameters:
    features (dict): A dictionary where keys are feature names (vote issues)
                     and values are 'yes' or 'no' representing the vote.

    Returns:
    tuple: A tuple containing:
           - str: The predicted party affiliation ('Democrat' or 'Republican').

        # Path 1: Leans Democrat (voted 'no' on aid to Nicaraguan contras)
        # Node 1: Secondary split - Vote on Export Administration Act - South Africa

            return 'Democrat', node_truth_values
