def predict_outcome_6m(age_years, body_mass_index_kg_m_2, gender_at_birth, ethnicity, education_age, work_status, smoke_status, hospital_stay_days, injury_severity_score_total, injury_severity_score_category, number_of_fractures, penetrating_injury, EQ_5D_5L, EQ_5D_VAS, SF36v2_mental, SF36v2_physical, HADS_anxiety, HADS_depression, TSK_11_total, PSEQ_total, IES_R_total, IES_R_avoidance, IES_R_hyperarousalm, IES_R_intrusion, pain_intensity_average, pain_intensity_current, pain_intensity_worst, pain_intensity_least, pain_region_count, pain_extent_percent, painDETECT_total, sleep_quality_last_24_hours, sleep_quality_since_injury, local_heat_pain_threshold, remote_heat_pain_threshold, local_cold_pain_threshold, remote_cold_pain_threshold, local_pressure_pain_threshold_kPa, remote_pressure_pain_threshold_kPa, C_reactive_protein_mg_L):
    """
    Predicts the pain and disability outcome at 6 months following musculoskeletal trauma
    as 'good' or 'bad' using a decision tree classifier with a maximum depth of 2.

    The decision tree is based on expert knowledge of key factors influencing long-term
    recovery from musculoskeletal trauma.

    Args:
        All features listed in the problem description, even if not used in this specific tree,
        are included in the signature to match the exact feature names provided.
        The features used for prediction are:
        - injury_severity_score_total (numeric): Total Injury Severity Score.
        - PSEQ_total (numeric): Pain Self-Efficacy Questionnaire total score (0-60). Higher is better.
        - HADS_anxiety (numeric): Hospital Anxiety and Depression Scale anxiety subscale score (0-21). Lower is better.

    Returns:
        tuple: A tuple containing:
            - str: The predicted outcome, either "good" or "bad".
            - list: A list of truth values for the inner nodes encountered on the prediction path.
                    Each entry is 1 if the node's condition is satisfied, and 0 otherwise.
                    The order reflects the path from root to leaf.
    """
    node_truth_values = []

    # Node 1 (Root): Splits based on Injury Severity Score Total
    # A score <= 15 generally indicates mild to moderate injury.
    if injury_severity_score_total <= 15:
        node_truth_values.append(1)  # Condition for Node 1 is met
        # Go to Left Child (Depth 2): PSEQ total
        # For mild/moderate injuries, psychosocial factors like self-efficacy are crucial for outcome.
        # PSEQ total >= 45 indicates relatively high self-efficacy (score range 0-60).
        if PSEQ_total >= 45:
            node_truth_values.append(1)  # Condition for Node 2 (Left Child) is met
            return "good", node_truth_values
        else:
            node_truth_values.append(0)  # Condition for Node 2 (Left Child) is not met
            return "bad", node_truth_values
    else:
        node_truth_values.append(0)  # Condition for Node 1 is not met
        # Go to Right Child (Depth 2): HADS anxiety
        # For more severe injuries, high anxiety is a strong predictor of poor outcome,
        # even if other factors are stable. Low anxiety might indicate better coping.
        # HADS anxiety <= 7 indicates absence of clinical anxiety (score range 0-21).
        if HADS_anxiety <= 7:
            node_truth_values.append(1)  # Condition for Node 3 (Right Child) is met
            return "good", node_truth_values
        else:
            node_truth_values.append(0)  # Condition for Node 3 (Right Child) is not met
            return "bad", node_truth_values
