def predict_survival(
    age_years: int,
    sex_male_female: str,
    steriods_yes_no: str,
    antivirals_yes_no: str,
    fatigue_yes_no: str,
    malaise_yes_no: str,
    anorexia_yes_no: str,
    liver_big_yes_no: str,
    liver_firm_yes_no: str,
    spleen_palpable_yes_no: str,
    spider_angiomata_yes_no: str,
    ascites_yes_no: str,
    varices_yes_no: str,
    bilirubin_mg_dL: float,
    alkaline_phosphatase_units_liter: float,
    serum_glutamic_oxaloacetic_transaminase_level_units_liter: float,
    albumin_level_g_dL: float,
    prothrombin_time_seconds: float,
    histologic_data_yes_no: str
):
    """
    Induces a decision tree classifier to predict patient survival based on key clinical features.
    The tree has a maximum depth of 2.

    Parameters:
    - age_years (int): Age of the patient in years.
    - sex_male_female (str): Sex of the patient ('male' or 'female').
    - steriods_yes_no (str): Whether the patient is on steroids ('yes' or 'no').
    - antivirals_yes_no (str): Whether the patient is on antivirals ('yes' or 'no').
    - fatigue_yes_no (str): Presence of fatigue ('yes' or 'no').
    - malaise_yes_no (str): Presence of malaise ('yes' or 'no').
    - anorexia_yes_no (str): Presence of anorexia ('yes' or 'no').
    - liver_big_yes_no (str): Whether the liver is enlarged ('yes' or 'no').
    - liver_firm_yes_no (str): Whether the liver is firm ('yes' or 'no').
    - spleen_palpable_yes_no (str): Whether the spleen is palpable ('yes' or 'no').
    - spider_angiomata_yes_no (str): Presence of spider angiomata ('yes' or 'no').
    - ascites_yes_no (str): Presence of ascites ('yes' or 'no').
    - varices_yes_no (str): Presence of varices ('yes' or 'no').
    - bilirubin_mg_dL (float): Bilirubin level in mg/dL.
    - alkaline_phosphatase_units_liter (float): Alkaline phosphatase level in units/liter.
    - serum_glutamic_oxaloacetic_transaminase_level_units_liter (float): SGOT level in units/liter.
    - albumin_level_g_dL (float): Albumin level in g/dL.
    - prothrombin_time_seconds (float): Prothrombin time in seconds.
    - histologic_data_yes_no (str): Presence of histologic data ('yes' or 'no').

    Returns:
    - tuple: A tuple containing the prediction ('Survive' or 'Die') and a list
             representing the truth values of the inner nodes (1 if condition
             satisfied, 0 otherwise).
    """
    truth_values = []

    # Node 1 (Root): age (years)
    # Condition: age_years <= 60
    if age_years <= 60:
        truth_values.append(1)  # Condition 'age_years <= 60' is True
        # Node 2 (Left child of Node 1): prothrombin time (seconds)
        # Condition: prothrombin_time_seconds > 14.0
        if prothrombin_time_seconds > 14.0:
            truth_values.append(1)  # Condition 'prothrombin_time_seconds > 14.0' is True
            # Leaf Node
            return "Die", truth_values
        else:  # prothrombin_time_seconds <= 14.0
            truth_values.append(0)  # Condition 'prothrombin_time_seconds > 14.0' is False
            # Leaf Node
            return "Survive", truth_values
    else:  # age_years > 60
        truth_values.append(0)  # Condition 'age_years <= 60' is False
        # Node 3 (Right child of Node 1): albumin level (g/dL)
        # Condition: albumin_level_g_dL <= 3.5
        if albumin_level_g_dL <= 3.5:
            truth_values.append(1)  # Condition 'albumin_level_g_dL <= 3.5' is True
            # Leaf Node
            return "Die", truth_values
        else:  # albumin_level_g_dL > 3.5
            truth_values.append(0)  # Condition 'albumin_level_g_dL <= 3.5' is False
            # Leaf Node
            return "Survive", truth_values
