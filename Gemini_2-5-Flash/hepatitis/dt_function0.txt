def classify_survival(age_years, sex, steroids, antivirals, fatigue, malaise, anorexia, liver_big, liver_firm, spleen_palpable, spider_angiomata, ascites, varices, bilirubin_mg_dl, alkaline_phosphatase_units_liter, sgot_units_liter, albumin_level_g_dl, prothrombin_time_seconds, histologic_data):
    """
    Classifies whether a patient survives or dies based on critical medical features.
    This decision tree has a maximum depth of 2.

    Args:
        age_years (int): Age of the patient in years.
        sex (str): 'male' or 'female'.
        steroids (str): 'yes' or 'no' if the patient is on steroids.
        antivirals (str): 'yes' or 'no' if the patient is on antivirals.
        fatigue (str): 'yes' or 'no' if the patient experiences fatigue.
        malaise (str): 'yes' or 'no' if the patient experiences malaise.
        anorexia (str): 'yes' or 'no' if the patient experiences anorexia.
        liver_big (str): 'yes' or 'no' if the liver is enlarged.
        liver_firm (str): 'yes' or 'no' if the liver is firm to palpation.
        spleen_palpable (str): 'yes' or 'no' if the spleen is palpable.
        spider_angiomata (str): 'yes' or 'no' if spider angiomata are present.
        ascites (str): 'yes' or 'no' if ascites is present.
        varices (str): 'yes' or 'no' if varices are present.
        bilirubin_mg_dl (float): Bilirubin level in mg/dL.
        alkaline_phosphatase_units_liter (float): Alkaline phosphatase level in units/liter.
        sgot_units_liter (float): Serum glutamic-oxaloacetic transaminase level (SGOT) in units/liter.
        albumin_level_g_dl (float): Albumin level in g/dL.
        prothrombin_time_seconds (float): Prothrombin time in seconds.
        histologic_data (str): 'yes' or 'no' if histologic data is available.

    Returns:
        tuple: A tuple containing:
            - str: The predicted outcome ('survives' or 'dies').
            - list: A list of integers representing the truth values of the inner nodes traversed (1 if condition satisfied, 0 otherwise).
    """
    node_truth_values = []

    # Root Node: Prothrombin time is a critical indicator of liver synthetic function.
    # A prolonged prothrombin time (e.g., > 18 seconds) suggests severe liver dysfunction.
    if prothrombin_time_seconds > 18:
        node_truth_values.append(1)  # Condition 'prothrombin_time_seconds > 18' is satisfied
        # First-level branch for prolonged PT:
        # Ascites is a major sign of decompensated liver disease, indicating poor prognosis.
        if ascites == 'yes':
            node_truth_values.append(1)  # Condition 'ascites == "yes"' is satisfied
            prediction = 'dies'  # Prolonged PT and ascites are strong indicators of death
        else:  # ascites == 'no'
            node_truth_values.append(0)  # Condition 'ascites == "yes"' is not satisfied
            prediction = 'dies'  # Even without ascites, severely prolonged PT (>18s) suggests critical liver failure
    else:  # prothrombin_time_seconds <= 18 (PT is relatively normal or mildly prolonged)
        node_truth_values.append(0)  # Condition 'prothrombin_time_seconds > 18' is not satisfied
        # First-level branch for normal/mildly prolonged PT:
        # Ascites remains a critical factor for prognosis even if PT is relatively better.
        if ascites == 'yes':
            node_truth_values.append(1)  # Condition 'ascites == "yes"' is satisfied
            prediction = 'dies'  # Ascites is a very severe sign, leading to death even with better PT
        else:  # ascites == 'no'
            node_truth_values.append(0)  # Condition 'ascites == "yes"' is not satisfied
            prediction = 'survives'  # Normal/mild PT and no ascites indicate a more stable condition

    return prediction, node_truth_values
