def predict_penguin_species(island, culmen_length_mm, culmen_depth_mm, flipper_length_mm, body_mass_g, sex):
    """
    Classifies the species of a penguin (Adelie / Chinstrap / Gentoo) based on physical features.
    
    Args:
        island (str): The island where the penguin was found (Biscoe, Dream, Torgersen).
        culmen_length_mm (float): Culmen length in millimeters.
        culmen_depth_mm (float): Culmen depth in millimeters.
        flipper_length_mm (float): Flipper length in millimeters.
        body_mass_g (float): Body mass in grams.
        sex (str): Sex of the penguin (male, female).
        
    Returns:
        tuple: A tuple containing the predicted species (str) and a list of 
               truth values for the inner nodes traversed (list of int, 1 if condition met, 0 otherwise).
    """
    inner_node_truth_values = []

    # Node 0: Split based on flipper length to distinguish Gentoo from others.
    # Gentoo penguins typically have significantly longer flippers.
    if flipper_length_mm > 206.0: 
        inner_node_truth_values.append(1) # Condition for Node 0 (flipper_length_mm > 206.0) is True
        prediction = "Gentoo"
    else:
        inner_node_truth_values.append(0) # Condition for Node 0 (flipper_length_mm > 206.0) is False
        
        # Node 1: Split based on culmen length to distinguish Chinstrap from Adelie.
        # Chinstrap penguins generally have a longer culmen than Adelie penguins.
        if culmen_length_mm > 44.0:
            inner_node_truth_values.append(1) # Condition for Node 1 (culmen_length_mm > 44.0) is True
            prediction = "Chinstrap"
        else:
            inner_node_truth_values.append(0) # Condition for Node 1 (culmen_length_mm > 44.0) is False
            prediction = "Adelie"

    return prediction, inner_node_truth_values
