def classify_solvency(features):
    """
    Classifies a company as 'Solvent' or 'Insolvent' based on financial features.

    Args:
        features (dict): A dictionary containing the following financial ratios:
            - 'Earnings Before Interest and Taxes/Total Assets'
            - 'Net Income/Total Capital'
            - 'Sales/Total Assets'
            - 'Earnings Before Interest and Taxes/Sales'
            - 'Net Income/Sales'
            - 'Working Capital/Total Assets'
            - 'Equity/Total Liabilities'
            - 'Equity/Total Assets'

    Returns:
        tuple: A tuple containing:
            - str: The predicted class ('Solvent' or 'Insolvent').
            - list: A list of truth values for the inner nodes traversed (1 if condition satisfied, 0 otherwise).
    """
    truth_values = []

    # Node 1: Check the overall financial structure and leverage.
    # A high Equity/Total Liabilities ratio indicates a strong equity base and low reliance on debt,
    # which is a primary indicator of solvency. A threshold of 0.75 suggests that equity is 75%
    # or more of total liabilities, indicating a healthy buffer.
    if features['Equity/Total Liabilities'] > 0.75:
        truth_values.append(1)  # Condition 1 is satisfied
        prediction = "Solvent"
    else:
        truth_values.append(0)  # Condition 1 is not satisfied
        # Node 2: If leverage is high (Equity/Total Liabilities <= 0.75),
        # then short-term liquidity becomes a critical factor.
        # A positive Working Capital/Total Assets ratio indicates that current assets
        # are sufficient to cover current liabilities, mitigating immediate default risk.
        # A threshold of 0.05 means working capital is at least 5% of total assets.
        if features['Working Capital/Total Assets'] > 0.05:
            truth_values.append(1)  # Condition 2 is satisfied
            prediction = "Solvent"
        else:
            truth_values.append(0)  # Condition 2 is not satisfied
            # If both leverage is high AND short-term liquidity is poor, the company is at high risk of insolvency.
            prediction = "Insolvent"

    return prediction, truth_values
