def classify_solvency(
    Earnings_Before_Interest_and_Taxes_Total_Assets: float,
    Net_Income_Total_Capital: float,
    Sales_Total_Assets: float,
    Earnings_Before_Interest_and_Taxes_Sales: float,
    Net_Income_Sales: float,
    Working_Capital_Total_Assets: float,
    Equity_Total_Liabilities: float,
    Equity_Total_Assets: float
) -> tuple[str, list[int]]:
    """
    Classifies a company as solvent or not_solvent based on financial ratios.

    Args:
        Earnings_Before_Interest_and_Taxes_Total_Assets (float): EBIT / Total Assets.
        Net_Income_Total_Capital (float): Net Income / Total Capital.
        Sales_Total_Assets (float): Sales / Total Assets.
        Earnings_Before_Interest_and_Taxes_Sales (float): EBIT / Sales.
        Net_Income_Sales (float): Net Income / Sales.
        Working_Capital_Total_Assets (float): Working Capital / Total Assets.
        Equity_Total_Liabilities (float): Equity / Total Liabilities.
        Equity_Total_Assets (float): Equity / Total Assets.

    Returns:
        tuple[str, list[int]]: A tuple containing the prediction ('solvent' or 'not_solvent')
                               and a list of truth values for the inner nodes.
                               The truth values are 1 if the condition is satisfied, 0 otherwise.
    """
    truth_values = []

    # Root Node: Equity/Total Assets - A fundamental measure of a company's financial structure
    # A higher ratio indicates a stronger equity base and lower reliance on debt.
    if Equity_Total_Assets > 0.4:  # Threshold: > 40% equity financing
        truth_values.append(1)  # Condition 'Equity_Total_Assets > 0.4' satisfied

        # Branch 1 (Equity_Total_Assets > 0.4): Company has a reasonable equity buffer.
        # Now check for short-term liquidity, as even financially strong companies can face liquidity issues.
        if Working_Capital_Total_Assets > 0.0:  # Threshold: Positive working capital
            truth_values.append(1)  # Condition 'Working_Capital_Total_Assets > 0.0' satisfied
            return "solvent", truth_values
        else:
            # Even with good equity, severe negative working capital signals immediate liquidity risk.
            truth_values.append(0)  # Condition 'Working_Capital_Total_Assets > 0.0' NOT satisfied
            return "not_solvent", truth_values
    else:
        truth_values.append(0)  # Condition 'Equity_Total_Assets > 0.4' NOT satisfied

        # Branch 2 (Equity_Total_Assets <= 0.4): Company has lower equity, indicating higher leverage.
        # Here, profitability is crucial to service debt and maintain solvency.
        if Earnings_Before_Interest_and_Taxes_Total_Assets > 0.05:  # Threshold: > 5% return on assets from operations
            truth_values.append(1)  # Condition 'Earnings_Before_Interest_and_Taxes_Total_Assets > 0.05' satisfied
            # Low equity but strong operational profitability can sustain the company.
            return "solvent", truth_values
        else:
            # Low equity combined with poor or low operational profitability is a strong indicator of distress.
            truth_values.append(0)  # Condition 'Earnings_Before_Interest_and_Taxes_Total_Assets > 0.05' NOT satisfied
            return "not_solvent", truth_values
