def classify_solvency(EBIT_TA, NI_TC, Sales_TA, EBIT_Sales, NI_Sales, WC_TA, Equity_TL, Equity_TA):
    """
    Classifies whether a company is solvent or insolvent based on key financial ratios.

    Parameters:
    EBIT_TA (float): Earnings Before Interest and Taxes / Total Assets
    NI_TC (float): Net Income / Total Capital
    Sales_TA (float): Sales / Total Assets
    EBIT_Sales (float): Earnings Before Interest and Taxes / Sales
    NI_Sales (float): Net Income / Sales
    WC_TA (float): Working Capital / Total Assets
    Equity_TL (float): Equity / Total Liabilities
    Equity_TA (float): Equity / Total Assets

    Returns:
    tuple: A tuple containing the prediction ("Solvent" or "Insolvent")
           and a list of truth values for the inner nodes encountered (1 if condition satisfied, 0 otherwise).
    """
    node_truth_values = []

    # Root Node (Depth 0): Assess overall financial structure and leverage
    # Condition: Is the company highly leveraged (Equity / Total Liabilities <= 0.75)?
    if Equity_TL <= 0.75:
        node_truth_values.append(1)  # Condition 'Equity_TL <= 0.75' is True
        # Inner Node (Depth 1, Branch 1): For highly leveraged companies, check profitability
        # Condition: Is the net profit margin very low or negative (Net Income / Sales <= 0.01)?
        if NI_Sales <= 0.01:
            node_truth_values.append(1)  # Condition 'NI_Sales <= 0.01' is True
            # Leaf Node (Depth 2): Highly leveraged and low/negative profitability -> Insolvent
            return "Insolvent", node_truth_values
        else:
            node_truth_values.append(0)  # Condition 'NI_Sales <= 0.01' is False
            # Leaf Node (Depth 2): Highly leveraged, but some profitability. Still a high risk for insolvency.
            return "Insolvent", node_truth_values
    else:
        node_truth_values.append(0)  # Condition 'Equity_TL <= 0.75' is False (i.e., Equity_TL > 0.75, stronger equity base)
        # Inner Node (Depth 1, Branch 2): For companies with a stronger equity base, check liquidity
        # Condition: Is working capital very low or negative (Working Capital / Total Assets <= 0.0)?
        if WC_TA <= 0.0:
            node_truth_values.append(1)  # Condition 'WC_TA <= 0.0' is True
            # Leaf Node (Depth 2): Strong equity, but immediate liquidity problems -> Insolvent
            return "Insolvent", node_truth_values
        else:
            node_truth_values.append(0)  # Condition 'WC_TA <= 0.0' is False
            # Leaf Node (Depth 2): Strong equity and healthy liquidity -> Solvent
            return "Solvent", node_truth_values
