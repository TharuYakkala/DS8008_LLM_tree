def classify_leaving_certificate(sex, drumcondra_score, educational_level, father_prestige_score, school_type):
    """
    Classifies whether a school leaving certificate is taken based on provided features.
    
    Args:
        sex (str): 'male' or 'female'.
        drumcondra_score (int): Drumcondra Verbal Reasoning Test Score.
        educational_level (int): Educational level attained (0 to 10).
        father_prestige_score (int): Prestige score for father's occupation (0 to 28).
        school_type (str): Type of school ('secondary', 'vocational', 'primary terminal leaver').
        
    Returns:
        tuple: A tuple containing:
               - int: The prediction (1 if certificate taken, 0 if not taken).
               - list: A list of integers representing the truth values of inner nodes 
                       (1 if condition satisfied, 0 otherwise).
    """
    node_truth_values = []
    
    # Node 1: Primary split on Drumcondra Verbal Reasoning Test Score (academic potential)
    # A score below 80 is considered low, indicating less likelihood of pursuing the leaving certificate.
    if drumcondra_score <= 80:
        node_truth_values.append(1) # Condition for Node 1 satisfied
        
        # Node 2 (Left branch): For students with lower academic potential, school type is crucial.
        # If already on a 'primary terminal leaver' track, it's a strong indicator against the leaving certificate.
        if school_type == 'primary terminal leaver':
            node_truth_values.append(1) # Condition for Node 2 satisfied
            prediction = 0 # Low academic score and in a primary terminal track -> Not Taken
        else:
            node_truth_values.append(0) # Condition for Node 2 not satisfied (i.e., secondary or vocational)
            prediction = 0 # Even if not primary terminal, a low academic score suggests not taking the certificate
    else:
        node_truth_values.append(0) # Condition for Node 1 not satisfied (Drumcondra score > 80, higher potential)
        
        # Node 3 (Right branch): For students with higher academic potential, socio-economic factors become important.
        # A father's occupation prestige below 14 suggests lower socio-economic support for prolonged education.
        if father_prestige_score < 14:
            node_truth_values.append(1) # Condition for Node 3 satisfied
            prediction = 0 # High academic score but lower socio-economic support -> Might not take certificate (e.g., enter workforce)
        else:
            node_truth_values.append(0) # Condition for Node 3 not satisfied (father_prestige_score >= 14)
            prediction = 1 # High academic score and higher socio-economic support -> Certificate Taken
            
    return prediction, node_truth_values
